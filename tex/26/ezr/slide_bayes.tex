# ---- 3. Bayes ----
def like(i: Col, x, prior: float) -> float: |\rc{1}|
  if isinstance(i, Sym):
    return (i.has.get(x, 0) + the.bayes.m * prior) / (i.n + the.bayes.m)
  v = i.div()
  if v == 0: return 1 if x == i.mu else 0
  nom = exp(-.5 * ((x - i.mu) / v)**2)
  return max(0, nom / (v * 2.5066 + 1E-32))

def classify(d: Data, row: Row) -> str: |\rc{2}|
  """Return the class with max likelihood for 'row'."""
  most, out = -1E32, None
  for k, klass in d.klasses.items():
    guess = likes(klass, row, len(d.rows), len(d.klasses))
    if guess > most:
      most, out = guess, k
  return out

def likes(d: Data, row: Row, nAll: int, nKlasses: int) -> float:
  prior = (len(d.rows) + the.bayes.k) / (nAll + the.bayes.k * nKlasses)
  out = log(prior)
  for c in d.cols.xs:
    if row[c.at] != "?":
      out += log(like(c, row[c.at], prior))
  return out
