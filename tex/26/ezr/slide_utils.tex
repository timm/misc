# ---- 0. Utilities ----
def csv(file: str) -> Iterable: |\rc{1}|
  """Reads a csv file, lazily yielding rows."""
  with open(file) as fp:
    for line in fp:
      row = [coerce(x) for x in line.strip().split(",")]
      yield row

def coerce(s: str) -> Any:
  try: return float(s)
  except: return s.strip()

def same(xs: list, ys: list) -> bool: |\rc{2}|
  """True if two lists are statistically indistinguishable."""
  return cliffsDelta(xs, ys) and ks(xs, ys)

def cliffsDelta(xs: list, ys: list) -> bool:
  n, gt, lt = 0, 0, 0
  for x in xs:
    for y in ys:
      n += 1
      if x > y: gt += 1
      if x < y: lt += 1
  return abs(lt - gt) / n <= the.stats.cliffs
