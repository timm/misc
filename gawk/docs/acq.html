import random, math
BIG = 1e32                     # handy “infinite” value
#

── tiny object helper ───────────────────────────────────────────────────────

class o:
# 
    def __init__(i, **d): i.__dict__.update(**d)
    __repr__ = lambda i: str(i.__dict__)
#

── column summaries ─────────────────────────────────────────────────────────

def Num(at=0, txt=""):
    return o(it=Num, n=0, at=at, txt=txt,
             mu=0, m2=0, sd=0, lo=BIG, hi=-BIG,
             heaven=(0 if txt.endswith("-") else 1))
# 
def Sym(at=0, txt=""):
    return o(it=Sym, n=0, at=at, txt=txt, has={})
#

── incremental add / sub (Welford for Nums) ───────────────────────────────

def _sym(sym,s,inc):
    sym.n += inc
    sym.has[s] = sym.has.get(s,0) + inc
# 
def _num(num,v,inc):
    num.n += inc
    num.lo, num.hi = min(v,num.lo), max(v,num.hi)
    d = v - num.mu
    if num.n > 0:
        num.mu += inc * d / num.n
        num.m2 += inc * d * (v - num.mu)
        num.sd  = 0 if num.n < 2 else (max(0,num.m2)/(num.n-1))**.5
# 
def add(i,v,inc=1,purge=False):
    if v == "?": return v
    if i.it in (Num, Sym):
        (_num if i.it is Num else _sym)(i, v, inc)
    else:                               # Data object
        if inc > 0: i._rows.append(v)
        elif purge: i._rows.remove(v)   # only on explicit purge
        i.n += inc
        for c in i.cols.all: add(c, v[c.at], inc)
    return v
# 
def sub(i,v,purge=False): return add(i, v, -1, purge)
#

── table-wide helpers ───────────────────────────────────────────────────────

def Cols(names):
    all,x,y = [],[],[]
    for c,s in enumerate(names):
        col = (Num if s[0].isupper() else Sym)(at=c, txt=s)
        all.append(col)
        if s[-1] != "X": (y if s[-1] in "+-" else x).append(col)
    return o(names=names, all=all, x=x, y=y)
# 
def Data(init):
    names, *rows = init
    d = o(it=Data, _rows=[], n=0, cols=Cols(names))
    for r in rows: add(d, r)
    return d
# 
def clone(data, rows): return Data([data.cols.names] + rows)
#

── tiny option bag (defaults) ───────────────────────────────────────────────

the = o(m=2, k=1, p=2, Assume=4, Build=20, Few=128,
        acq="xploit", guess=0.5)
#

── distances & normalisation ───────────────────────────────────────────────

minkowski = lambda xs: (sum(x**the.p for x in xs) /
                        (len(xs) + 1/BIG)) ** (1/the.p)
norm      = lambda num,v: v if v=="?" else (v-num.lo)/(num.hi-num.lo+1/BIG)
ydist     = lambda d,r: minkowski(abs(norm(c,r[c.at]) - c.heaven)
                                 for c in d.cols.y)
ydists    = lambda d: sorted(d._rows, key=lambda r: ydist(d, r))
#

── Naive-Bayes “likeness” ───────────────────────────────────────────────────

def pdf(col,v,prior=0):
    if col.it is Sym:
        return (col.has.get(v,0) + the.m*prior) / (col.n + the.m + 1/BIG)
    sd = col.sd or 1/BIG
    var = 2*sd*sd
    return min(1, math.exp(-((v-col.mu)**2)/var) /
                  (2*math.pi*var)**0.5)
# 
def like(data,row,nall=2,nh=100):
    prior = (data.n + the.k) / (nall + the.k*nh)
    s = math.log(prior)
    for c in data.cols.x:
        v = row[c.at]
        if v != "?": s += math.log(pdf(c, v, prior))
    return s
#

── main active-learning loop ────────────────────────────────────────────────

def acquires(data, start=None, stop=None, guess=None, few=None):
# 
    def _acq(b,r,acq="xploit",p=1):
        b,r = math.e**b, math.e**r
        q   = 0 if acq=="xploit" else (1 if acq=="xplor" else 1-p)
        return (b + r*q) / abs(b*q - r + 1/BIG)
# 
    def _guess(row):        # heuristic score for a candidate row
        return _acq(like(best,row,n,2), like(rest,row,n,2),
                    the.acq, n / the.Build)
#

user-tunable knobs (fallback to the)

    start = the.Assume if start is None else start
    stop  = the.Build  if stop  is None else stop
    guess = the.guess  if guess is None else guess
    few   = the.Few    if few   is None else few

    random.shuffle(data._rows)
    n        = start
    todo     = data._rows[n:]
    bestrest = clone(data, data._rows[:n])
    done     = ydists(bestrest)
    cut      = round(n**guess)
    best     = clone(data, done[:cut])
    rest     = clone(data, done[cut:])

    while len(todo) > 2 and n < stop:
        n += 1
        hi, *lo = sorted(todo[:few*2], key=_guess, reverse=True)
        todo    = lo[:few] + todo[few*2:] + lo[few:]
        add(bestrest, add(best, hi))
        best._rows = ydists(bestrest)
        if len(best._rows) >= round(n**guess):
            add(rest, sub(best, best._rows.pop(-1)))

    return o(best=best, rest=rest, test=todo)
#
