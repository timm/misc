#
#

Define your percentages

passed=23
coverage=80
lint=92
#

Determine color based on percentage

if [ "$passed" -gt 50 ]; then
    passed_color="brightgreen"
else
    passed_color="red"
fi

if [ "$coverage" -gt 80 ]; then
    coverage_color="brightgreen"
else
    coverage_color="yellow"
fi

if [ "$lint" -gt 80 ]; then
    lint_color="brightgreen"
else
    lint_color="yellow"
fi
#

Generate the badge URLs

passed_badge="https://img.shields.io/badge/passed-${passed}%-${passed_color}-flat-square"
coverage_badge="https://img.shields.io/badge/coverage-${coverage}%-${coverage_color}-flat-square"
lint_badge="https://img.shields.io/badge/lint-${lint}%-${lint_color}-flat-square"
#

Print the badge HTML for README

echo "![Passed]($passed_badge)"
echo "![Coverage]($coverage_badge)"
echo "![Lint]($lint_badge)"
