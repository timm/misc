from types import SimpleNamespace as o
from random import shuffle, sample as some
import math
# 
def Core():
    return o(n=0, sum={}, freq={}, mid=None, rows=[])
# 
def Cols(row):
    cols = list(row.keys())
    nums = {col for col in cols if col[0].isupper()}
    goals = {col for col in nums if col.endswith(('+', '-'))}
    heaven = {col: (1 if col.endswith('+') else 0) for col in goals}
    span = o(lo={}, hi={})
    return o(nums=nums, goals=goals, heaven=heaven, cols=cols, span=span)
# 
def ydist(row):
    return math.sqrt(sum((norm(row[col], col) - meta.heaven[col])**2 for col in meta.goals))
# 
def norm(val, col):
    return (val - meta.span.lo[col]) / (meta.span.hi[col] - meta.span.lo[col] + 1e-32)
# 
def add(id1, row, delta=1):
    c = getattr(cores, id1)
    c.n += delta
    c.mid = None
    
    for col, val in row.items():
        if col in meta.nums:
            if delta == 1:
                meta.span.lo[col] = min(meta.span.lo.get(col, val), val)
                meta.span.hi[col] = max(meta.span.hi.get(col, val), val)
            c.sum[col] = c.sum.get(col, 0) + val * delta
        else:
            c.freq[col] = c.freq.get(col, {})
            c.freq[col][val] = c.freq[col].get(val, 0) + delta
    
    if delta == 1:
        c.rows.append(row)
    else:
        c.rows.remove(row)
    return row
# 
def sub(id1, row):
    add(id1, row, -1)
    return row
# 
def mid(id1):
    c = getattr(cores, id1)
    if c.mid is None:
        c.mid = {col: (c.sum[col] / c.n if col in meta.nums 
                      else max(c.freq[col], key=c.freq[col].get) if col in c.freq and c.freq[col] else None)
                for col in set(c.sum.keys()) | set(c.freq.keys())} if c.n > 0 else {}
    return c.mid
# 
def dist_to_mid(row, centroid):
    return sum((row[col] - centroid.get(col, 0))**2 for col in meta.nums if col in centroid)**0.5
# 
def likely(rows, budget=40, start=4):
    global cores
    cores = o(best=Core(), rest=Core(), labeled=Core(), unlabeled=Core())
#

Initialize all as unlabeled (builds span automatically)

    [add('unlabeled', r) for r in rows]
#

Move random start to labeled

    shuffle(cores.unlabeled.rows)
    for _ in range(start):
        add('labeled', sub('unlabeled', cores.unlabeled.rows[0]))
#

Copy labeled to best/rest by ydist

    sorted_labeled = sorted(cores.labeled.rows, key=ydist)
    [add('best', r) for r in sorted_labeled[:int(start**0.5)]]
    [add('rest', r) for r in sorted_labeled[int(start**0.5):]]
#

Active learning: pick rows nearer to best than rest

    while cores.unlabeled.rows and cores.labeled.n < budget:
#

Find first row closer to best

        guess = 0
        for i, row in enumerate(some(cores.unlabeled.rows, min(20, len(cores.unlabeled.rows)))):
            if dist_to_mid(row, mid('best')) < dist_to_mid(row, mid('rest')):
                guess = i; break
#

Move through pipeline: unlabeled -> labeled -> best

        found = cores.unlabeled.rows[guess]
        add('best', add('labeled', sub('unlabeled', found)))
#

Rebalance: move worst from best to rest

        if cores.best.n > cores.labeled.n**0.5:
            add('rest', sub('best', max(cores.best.rows, key=ydist)))
    
    return sorted(cores.labeled.rows, key=ydist)
#

Usage

data = [{'name': f'row{i}', 'Age': 20+i, 'Income+': 40000+i*1000, 'Pain-': i%10} 
        for i in range(100)]
header, *rows = [data[0]], data[1:]
meta = Cols(header[0])
cores = None

if __name__ == "__main__":
    result = likely(data, budget=20)
    print(f"Found {len(result)} examples, best ydist: {ydist(result[0]):.3f}")
